## 快速测试脚本

打开Chrome DevTools Console，运行以下代码:

```javascript
// 测试所有按钮尺寸
document.querySelectorAll('.action-btn, .result-btn, .feedback-close, #ready-btn').forEach(btn => {
  const rect = btn.getBoundingClientRect();
  console.log(btn.className, 'height:', rect.height, 'width:', rect.width);
  if (rect.height < 44) {
    console.warn('❌ 按钮高度不足:', btn.className);
  } else {
    console.log('✅ 按钮高度合格:', btn.className);
  }
});

// 测试viewport
console.log('Viewport width:', window.innerWidth);
console.log('Viewport height:', window.innerHeight);

// 测试底部留白
const body = document.body;
const bodyStyles = window.getComputedStyle(body);
console.log('Body padding-bottom:', bodyStyles.paddingBottom);

// 测试媒体查询
const mediaQueries = [
  '(max-width: 768px)',
  '(max-width: 480px)',
  '(max-width: 360px)',
  '(max-width: 768px) and (orientation: landscape)'
];

mediaQueries.forEach(query => {
  const matches = window.matchMedia(query).matches;
  console.log(query, ':', matches ? '✅ 匹配' : '❌ 不匹配');
});
```
